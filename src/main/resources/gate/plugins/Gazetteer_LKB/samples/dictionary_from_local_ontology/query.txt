prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#>
SELECT ?la ?entity ?cl
WHERE {
   ?entity rdfs:label ?la .
   ?entity a ?cl .
   ?cl rdfs:subClassOf rdfs:Class .
   OPTIONAL
   {
       ?sc rdfs:subClassOf ?cl.
       ?entity a ?sc .
       filter(?cl != ?sc)      
   }   
   filter (!bound(?sc) && isURI(?cl))
}
