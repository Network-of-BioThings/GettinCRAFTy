104      /**
105       * Parse the current document.  (This is the principal 
106       * method called by a CorpusController.)
107       */
108      public void execute() throws ExecutionException {
109        interrupted = false;
110        long startTime = System.currentTimeMillis();
111        if(document == null) {
112          throw new ExecutionException("No document to process!");
113        }
114        fireStatusChanged("Running " + this.getName() + " on " + document.getName());
115        fireProgressChanged(0);
116    
117        if (debugMode) {
118          System.out.println("Parsing document: " + document.getName());
119        }
120    
121        if (useMapping && (! mappingLoaded) ) {
122          System.err.println("Warning: no mapping loaded!");
123        }
124        
125        checkInterruption();
126        if (addConstituentAnnotations || addDependencyFeatures || addDependencyAnnotations || addPosTags) {
127          parseSentences(document.getAnnotations(annotationSetName));
128        }
129        else {
130          System.err.println("There is nothing for the parser to do.");
131          System.err.println("Please enable at least one of the \"add...\" options.");
132        }
133        
134        fireProcessFinished();
135        fireStatusChanged("Finished " + this.getName() + " on " + document.getName()
136            + " in " + NumberFormat.getInstance().format(
137                (double)(System.currentTimeMillis() - startTime) / 1000)
138                + " seconds!");
139      }

142      /**
143       * Initialize the Parser resource.  In particular, load the trained data
144       * file.
145       */
146      public Resource init() throws ResourceInstantiationException {
147        instantiateStanfordParser();
148        if (mappingFile != null) {
149          loadTagMapping(mappingFile);
150        }
151    
152        super.init();
153        
154        if(tlppClass == null || tlppClass.equals("")) {
155          throw new ResourceInstantiationException(
156                  "TLPP class name must be specified");
157        }
158        try {
159          Class<?> tlppClassObj =
160                  Class.forName(tlppClass, true, Gate.getClassLoader());
161          if(!TreebankLangParserParams.class.isAssignableFrom(tlppClassObj)) {
162            throw new ResourceInstantiationException(tlppClassObj
163                    + " does not implement "
164                    + TreebankLangParserParams.class.getName());
165          }
166          TreebankLangParserParams tlpp =
167                  TreebankLangParserParams.class.cast(tlppClassObj.newInstance());
168          gsf = tlpp.treebankLanguagePack().grammaticalStructureFactory();
169        }
170        catch(UnsupportedOperationException e) {
171          throw new ResourceInstantiationException(e);
172        }
173        catch(ClassNotFoundException e) {
174          throw new ResourceInstantiationException("Class " + tlppClass
175                  + " not found", e);
176        }
177        catch(InstantiationException e) {
178          throw new ResourceInstantiationException("Error creating TLPP object", e);
179        }
180        catch(IllegalAccessException e) {
181          throw new ResourceInstantiationException("Error creating TLPP object", e);
182        }
183        return this;
184      }

199      /**
200       * Find all the Sentence annotations and iterate through them, parsing one
201       * sentence at a time and storing the result in the output AS. (Sentences are
202       * scanned for Tokens. You have to run the ANNIE tokenizer and splitter before
203       * this PR.)
204       * @throws ExecutionInterruptedException 
205       */
206      private void parseSentences(AnnotationSet annotationSet) throws ExecutionInterruptedException { 
207        List<Annotation> sentences = gate.Utils.inDocumentOrder(annotationSet.get(inputSentenceType));
208        int sentencesDone = 0;
209        int nbrSentences = sentences.size();
210    
211        for (Annotation sentence : sentences) {
212          parseOneSentence(annotationSet, sentence, sentencesDone, nbrSentences);
213          sentencesDone++;
214          checkInterruption();
215        }
216        
217        sentencesDone++;
218        fireProgressChanged((int)(100 * sentencesDone / nbrSentences));
219    
220       }

224      /**
225       * Generate the special data structure for one sentence and pass the List of
226       * Word to the parser.  Apply the annotations back to the document.
227       * 
228       * @param sentence
229       *          the Sentence annotation
230       * @param s
231       *          sentence number of debugging output
232       * @param ofS
233       *          total number of sentences for debugging output
234       * @return  null if the sentence is empty
235       * @throws ExecutionInterruptedException 
236       */
237      private void parseOneSentence(AnnotationSet annotationSet, Annotation sentence, int sentCtr, int sentCount) throws ExecutionInterruptedException {
238        Tree tree;
239        
240        StanfordSentence stanfordSentence = new StanfordSentence(sentence, inputTokenType, annotationSet, reusePosTags);
241        if (debugMode) {
242          System.out.println(stanfordSentence.toString());
243        }
244    
245        /* Ignore an empty Sentence (sometimes the regex splitter can create one
246         * with no Token annotations in it).
247         */
248        if ( stanfordSentence.isNotEmpty() ) {
249          List<Word> wordList = stanfordSentence.getWordList();
250    
251          if (reusePosTags) {
252            int nbrMissingTags = stanfordSentence.numberOfMissingPosTags();
253            if (nbrMissingTags > 0)  {
254              double percentMissing = Math.ceil(100.0 * ((float) nbrMissingTags) /
255                      ((float) stanfordSentence.numberOfTokens()) );
256              System.err.println("Warning (sentence " + sentCtr + "): " + (int) percentMissing 
257                      + "% of the Tokens are missing POS tags." );
258            }
259          }
260    
261          tree = stanfordParser.parseTree(wordList); 
262          checkInterruption();
263    
264          if (addConstituentAnnotations || addPosTags) {
265            annotatePhraseStructureRecursively(annotationSet, stanfordSentence, tree, tree);
266          }
267    
268          checkInterruption();
269          if (addDependencyFeatures || addDependencyAnnotations) {
270            annotateDependencies(annotationSet, stanfordSentence, tree);
271          }
272    
273          if (debugMode) {
274            System.out.println("Parsed sentence " + sentCtr + " of " + sentCount);
275          }
276        }
277        
278        else if (debugMode) {
279          System.out.println("Ignored empty sentence " + sentCtr + " of " + sentCount);
280        }
281      }

284      /**
285       * Generate a SyntaxTreeNode Annotation corresponding to this Tree.  Work 
286       * recursively so that the annotations are actually generated from the 
287       * bottom up, in order to build the consists list of annotation IDs.
288       * 
289       * @param tree  the current subtree
290       * @param rootTree  the whole sentence, used to find the span of the current subtree
291       * @return a GATE Annotation of type "SyntaxTreeNode"
292       */
293      protected Annotation annotatePhraseStructureRecursively(AnnotationSet annotationSet, StanfordSentence stanfordSentence, Tree tree, Tree rootTree) {
294        Annotation annotation = null;
295        Annotation child;
296        String label   = tree.value();
297    
298        List<Tree> children = tree.getChildrenAsList();
299    
300        if (children.size() == 0) {
301          return null;
302        }
303        /* implied else */
304    
305        /* following line generates ClassCastException
306         *          IntPair span = tree.getSpan();
307         * edu.stanford.nlp.ling.CategoryWordTag
308         * at edu.stanford.nlp.trees.Tree.getSpan(Tree.java:393)
309         * but I think it's a bug in the parser, so I'm hacking 
310         * around it as follows. */
311        int startPos = Trees.leftEdge(tree, rootTree);
312        int endPos   = Trees.rightEdge(tree, rootTree);
313        
314        Long startNode = stanfordSentence.startPos2offset(startPos);
315        Long endNode   = stanfordSentence.endPos2offset(endPos);
316    
317        List<Integer> consists = new ArrayList<Integer>();
318    
319        Iterator<Tree> childIter = children.iterator();
320        while (childIter.hasNext()) {
321          child = annotatePhraseStructureRecursively(annotationSet, stanfordSentence, childIter.next(), rootTree);
322          if  ( (child != null)  &&
323            (! child.getType().equals(inputTokenType) )) {
324            consists.add(child.getId());
325          }
326        }
327        annotation = annotatePhraseStructureConstituent(annotationSet, startNode, endNode, label, consists, tree.depth());
328    
329        return annotation;
330      }

334      /**
335       * Record one constituent as an annotation.
336       * 
337       * @param startOffset
338       * @param endOffset
339       * @param label
340       * @param consists
341       * @param depth
342       * @return
343       */
344      private Annotation annotatePhraseStructureConstituent(AnnotationSet annotationSet, Long startOffset, Long endOffset, String label, 
345        List<Integer> consists, int depth) {
346        Annotation phrAnnotation = null;
347        Integer phrID;
348    
349        try {
350          String cat;
351          if (useMapping && mappingLoaded) {
352            cat  = translateTag(label);
353          }
354          else {
355            cat = label; 
356          }
357          
358          if (addConstituentAnnotations) {
359            String text = document.getContent().getContent(startOffset, endOffset).toString();
360            FeatureMap fm = gate.Factory.newFeatureMap();
361            fm.put(PHRASE_CAT_FEATURE, cat);
362            fm.put("text", text);
363    
364            /* Ignore empty list features on the token-equivalent annotations. */
365            if (consists.size() > 0) {
366              fm.put("consists", consists);
367            }
368    
369            phrID  = annotationSet.add(startOffset, endOffset, PHRASE_ANNOTATION_TYPE, fm);
370            phrAnnotation = annotationSet.get(phrID);
371            recordID(annotationSet, phrID);
372          }
373    
374          if ( addPosTags && (depth == 1) ) {
375            /* Expected to be a singleton set! */
376            AnnotationSet tokenSet = annotationSet.get(inputTokenType, startOffset, endOffset);
377            if (tokenSet.size() == 1) {
378              Annotation token = tokenSet.iterator().next();
379    
380              /* Add POS tag to token.  
381               * (Note: GATE/Hepple uses "(" and ")" for Penn/Stanford's
382               * "-LRB-" and "-RRB-". */
383              String hepCat = StanfordSentence.unescapePosTag(cat);
384              token.getFeatures().put(POS_TAG_FEATURE, hepCat);
385              
386            }
387            else {
388              System.err.println("Found a tokenSet with " + tokenSet.size() + " members!");
389            }
390          }
391        }
392        catch (InvalidOffsetException e) {
393          e.printStackTrace();
394        }
395        
396        return phrAnnotation;
397      }

401      @SuppressWarnings("unchecked")
402      private void annotateDependencies(AnnotationSet annotationSet, StanfordSentence stanfordSentence, Tree tree) {
403        GrammaticalStructure gs = gsf.newGrammaticalStructure(tree);
404        Collection<TypedDependency> dependencies = DependencyMode.getDependencies(gs, dependencyMode, includeExtraDependencies);
405    
406        if (dependencies == null) {
407          if (debugMode) {
408            System.out.println("dependencies == null");
409          }
410          return;
411        }
412        
413        String dependencyKind;
414        FeatureMap depFeatures;
415        Integer dependentTokenID, governorTokenID;
416        List<Integer> argList;
417        Long offsetLH0, offsetRH0, offsetLH1, offsetRH1, depLH, depRH;
418        Annotation governor, dependent;
419    
420        for(TypedDependency dependency : dependencies) {
421          if(debugMode) {
422            System.out.println(dependency);
423          }
424    
425          int governorIndex = dependency.gov().label().index() - 1;
426          governor  = stanfordSentence.startPos2token(governorIndex);
427          
428          int dependentIndex = dependency.dep().label().index() - 1;
429          dependent = stanfordSentence.startPos2token(dependentIndex);
430    
431          dependencyKind = dependency.reln().toString();
432          governorTokenID = governor.getId();
433          dependentTokenID = dependent.getId();
434          
435          if (addDependencyFeatures) {
436            List<DependencyRelation> depsForTok =
437              (List<DependencyRelation>) governor.getFeatures().get(dependenciesFeature);
438            
439            if(depsForTok == null) {
440              depsForTok = new ArrayList<DependencyRelation>();
441              governor.getFeatures().put(dependenciesFeature, depsForTok);
442            }
443            
444            depsForTok.add(new DependencyRelation(dependencyKind, dependentTokenID));
445          }
446          
447          if (addDependencyAnnotations) {
448            depFeatures = gate.Factory.newFeatureMap();
449            argList = new ArrayList<Integer>();
450            argList.add(governorTokenID);
451            argList.add(dependentTokenID);
452            depFeatures.put(DEPENDENCY_ARG_FEATURE, argList);
453            depFeatures.put(DEPENDENCY_LABEL_FEATURE, dependencyKind);
454            
455            offsetLH0 = governor.getStartNode().getOffset();
456            offsetRH0 = governor.getEndNode().getOffset();
457            offsetLH1 = dependent.getStartNode().getOffset();
458            offsetRH1 = dependent.getEndNode().getOffset();
459            
460            depLH = Math.min(offsetLH0, offsetLH1);
461            depRH = Math.max(offsetRH0, offsetRH1);
462            
463            try {
464              annotationSet.add(depLH, depRH, DEPENDENCY_ANNOTATION_TYPE, depFeatures);
465            }
466            catch(InvalidOffsetException e) {
467              e.printStackTrace();
468            }  
469          }
470        }
471      }

475      private void instantiateStanfordParser()
476        throws ResourceInstantiationException {
477        if(stanfordParser != null) return;
478        
479        try {
480          String filepath = Files.fileFromURL(parserFile).getAbsolutePath();
481          stanfordParser = LexicalizedParser.getParserFromSerializedFile(filepath);
482        }
483        catch(Exception e) {
484          throw new ResourceInstantiationException(e);
485        }
486      }

489      private void loadTagMapping(File mappingFile)  { 
490        tagMap = new HashMap<String, String>();
491        mappingLoaded = false;
492    
493        try {
494          if (mappingFile.exists() && mappingFile.canRead()) {
495    
496            BufferedReader br = new BufferedReader(new FileReader(mappingFile));
497            String line = "";
498    
499            // read until it reaches to an end of the file
500            while((line = br.readLine()) != null) {
501              // two columns delimited by whitespace 
502              String [] data = line.split("\\s+", 2);
503    
504              // are there key and value available
505              if(data == null || data.length < 2) {
506                continue;
507              } else {
508                // and add it to the map
509                tagMap.put(data[0].trim(), data[1].trim());
510              }
511            }
512    
513            br.close();
514          }
515    
516          else {
517            System.err.println("Can't find or read mapping file " 
518              + mappingFile.getPath() + " so no mappings will be used.");
519          }
520        } 
521        catch(Exception e) {
522          System.err.println("Exception trying to load mapping file "
523            + mappingFile.getPath());
524          e.printStackTrace();
525        }
526    
527        int nbrMapped = tagMap.size();
528        System.out.println("Loaded " + nbrMapped + " mappings from file " + mappingFile);
529        mappingLoaded = (nbrMapped > 0);
530      }

547        if(isInterrupted()) { throw new ExecutionInterruptedException(
548            "Execution of " + this.getName() + " has been abruptly interrupted!"); }
549      }

552      /**
553       * Translate the tag in the map, or leave it the same if there is no
554       * translation.
555       * 
556       * @param stanfordTag
557       * @return
558       */
559      private String translateTag(String stanfordTag) {
560        String translatedTag = stanfordTag;
561    
562        if (tagMap.containsKey(stanfordTag)) {
563          translatedTag = tagMap.get(stanfordTag);
564        }
565        
566        return translatedTag;
567      }
